            // ===== FUNÇÃO: CRIAR CAMPO =====

            function createFieldGroup(id, labelText, type, placeholder, required) {

                const group = document.createElement('div');

                group.className = 'form-group';

 

                const label = document.createElement('label');

                label.htmlFor = id;

                label.innerHTML = labelText + (required ? ' <span class="required">*</span>' : '');

 

                const input = document.createElement(type === 'textarea' ? 'textarea' : 'input');

                input.id = id;

                input.className = 'form-control';

                if (type !== 'textarea') {

                    input.type = type;

                    input.placeholder = placeholder || '';

                } else {

                    input.placeholder = placeholder || '';

                }

                if (required) input.required = true;

                if (type === 'number') {

                    input.step = 'any';

                    input.min = '0';

                }

 

                group.appendChild(label);

                group.appendChild(input);

                return group;

            }

 

            // ===== FUNÇÃO: CRIAR SELECT =====

            function createSelectGroup(id, labelText, options, required) {

                const group = document.createElement('div');

                group.className = 'form-group';

 

                const label = document.createElement('label');

                label.htmlFor = id;

                label.innerHTML = labelText + (required ? ' <span class="required">*</span>' : '');

 

                const select = document.createElement('select');

                select.id = id;

                select.className = 'form-control';

                if (required) select.required = true;

 

                options.forEach(opt => {

                    const option = document.createElement('option');

                    option.value = opt.value;

                    option.textContent = opt.label;

                    select.appendChild(option);

                });

 

                group.appendChild(label);

                group.appendChild(select);

                return group;

            }

 

            // ===== FUNÇÃO: CRIAR CAMPO DE FOTO =====

            function createFileGroup(id, labelText) {

                const group = document.createElement('div');

                group.className = 'form-group';

 

                const label = document.createElement('label');

                label.htmlFor = id;

                label.textContent = labelText;

 

                const wrapper = document.createElement('div');

                wrapper.className = 'file-input-wrapper';

 

                const fileLabel = document.createElement('div');

                fileLabel.className = 'file-label';

                fileLabel.innerHTML = `📷 Clique para selecionar uma foto<br><span style="font-size:12px;color:#94a3b8;">Formatos: JPG, PNG, WEBP</span>`;

 

                const input = document.createElement('input');

                input.type = 'file';

                input.id = id;

                input.accept = 'image/*';

                input.addEventListener('change', function() {

                    if (this.files && this.files[0]) {

                        const nameSpan = document.createElement('span');

                        nameSpan.className = 'file-name';

                        nameSpan.textContent = ` 📎 ${this.files[0].name}`;

                        fileLabel.innerHTML = `📷 Foto selecionada: `;

                        fileLabel.appendChild(nameSpan);

                    }

                });

 

                wrapper.appendChild(input);

                wrapper.appendChild(fileLabel);

                group.appendChild(label);

                group.appendChild(wrapper);

                return group;

            }

 

            // ===== FUNÇÃO: CALCULAR EQUIPAMENTO =====

            function calculateEquipment(index) {

                const powerInput = document.getElementById(`equip_power_${index}`);

                const timeInput = document.getElementById(`equip_time_${index}`);

                const qtyInput = document.getElementById(`equip_quantity_${index}`);

                const consumoEl = document.getElementById(`consumo_${index}`);

                const custoEl = document.getElementById(`custo_${index}`);

 

                if (!powerInput || !timeInput || !qtyInput || !consumoEl || !custoEl) return;

 

                const power = parseFloat(powerInput.value) || 0;

                const time = parseFloat(timeInput.value) || 0;

                const qty = parseFloat(qtyInput.value) || 1;

                const tariff = parseFloat(tariffSelect.value) || 0.92;

 

                // CONSUMO = POTÊNCIA * TEMPO * QUANTIDADE / 1000 (kWh)

                const consumo = (power * time * qty) / 1000;

                const custo = consumo * tariff;

 

                consumoEl.textContent = consumo.toFixed(3);

                custoEl.textContent = `R$ ${custo.toFixed(2)}`;

 

                // Atualiza resumo

                updateSummary();

            }

 

            // ===== FUNÇÃO: ATUALIZAR RESUMO =====

            function updateSummary() {

                const sections = document.querySelectorAll('.equipment-section');

                const tariff = parseFloat(tariffSelect.value) || 0.92;

                displayTariff.textContent = `R$ ${tariff.toFixed(2)} / kWh`;

 

                let totalConsumo = 0;

                let totalCusto = 0;

                let rows = '';

 

                sections.forEach((section, idx) => {

                    const num = idx + 1;

                    const nameEl = document.getElementById(`equip_name_${num}`);

                    const powerEl = document.getElementById(`equip_power_${num}`);

                    const qtyEl = document.getElementById(`equip_quantity_${num}`);

                    const timeEl = document.getElementById(`equip_time_${num}`);

 

                    const name = nameEl ? nameEl.value.trim() : '---';

                    const power = parseFloat(powerEl?.value) || 0;

                    const qty = parseFloat(qtyEl?.value) || 1;

                    const time = parseFloat(timeEl?.value) || 0;

                    const consumo = (power * time * qty) / 1000;

                    const custo = consumo * tariff;

 

                    totalConsumo += consumo;

                    totalCusto += custo;

 

                    if (name && power > 0) {

                        rows += `

                            <tr>

                                <td><strong>${name}</strong></td>

                                <td class="text-right">${power}</td>

                                <td class="text-center">${qty}</td>

                                <td class="text-right">${time.toFixed(1)}</td>

                                <td class="text-right">${consumo.toFixed(3)}</td>

                                <td class="text-right">R$ ${custo.toFixed(2)}</td>

                            </tr>

                        `;

                    }

                });

 

                summaryBody.innerHTML = rows || `

                    <tr>

                        <td colspan="6" style="text-align:center; color:#94a3b8; padding:20px;">

                            Nenhum equipamento cadastrado. Adicione um equipamento para ver o resumo.

                        </td>

                    </tr>

                `;

 

                summaryFooter.innerHTML = `

                    <tr>

                        <td colspan="3" class="total-label">TOTAL GERAL</td>

                        <td class="text-right"></td>

                        <td class="text-right" style="font-size:16px;">${totalConsumo.toFixed(3)} kWh</td>

                        <td class="text-right" style="font-size:16px; color:#0c5c64;">R$ ${totalCusto.toFixed(2)}</td>

                    </tr>

                `;

            }

 

            // ===== FUNÇÃO: ATUALIZAR ÍNDICES =====

            function updateIndexes() {

                const sections = document.querySelectorAll('.equipment-section');

                sections.forEach((section, idx) => {

                    const num = idx + 1;

                    section.dataset.index = num;

                    const numberSpan = section.querySelector('.section-title .number');

                    if (numberSpan) numberSpan.textContent = num;

                    const titleText = section.querySelector('.section-title');

                    if (titleText) {

                        const textNode = titleText.childNodes[2];

                        if (textNode) textNode.textContent = ` Equipamento ${num}`;

                    }

                    // Atualiza IDs

                    const inputs = section.querySelectorAll('input, select, textarea');

                    inputs.forEach(input => {

                        const oldId = input.id;

                        if (oldId) {

                            const parts = oldId.split('_');

                            if (parts.length >= 3) {

                                const newId = parts[0] + '_' + parts[1] + '_' + num;

                                input.id = newId;

                                const label = section.querySelector(`label[for="${oldId}"]`);

                                if (label) label.htmlFor = newId;

                            }

                        }

                    });

                    // Atualiza IDs dos resultados

                    const consumoEl = document.getElementById(`consumo_${idx + 1}`);

                    const custoEl = document.getElementById(`custo_${idx + 1}`);

                    if (consumoEl) consumoEl.id = `consumo_${num}`;

                    if (custoEl) custoEl.id = `custo_${num}`;

                    const resultBox = section.querySelector('.result-box');

                    if (resultBox) resultBox.id = `result_${num}`;

                });

                updateSummary();

            }

            function resetForm() {

                if (!confirm('Tem certeza que deseja limpar todos os dados?')) return;

 

                document.getElementById('studentName').value = '';

                document.getElementById('studentClass').value = '';

                document.getElementById('collectionDate').value = '';

                document.getElementById('collectionLocation').value = '';

                tariffSelect.selectedIndex = 0;

 

                const sections = document.querySelectorAll('.equipment-section');

                sections.forEach((section, idx) => {

                    if (idx > 0) {

                        section.remove();

                    } else {

                        const inputs = section.querySelectorAll('input, select, textarea');

                        inputs.forEach(input => {

                            if (input.type === 'file') {

                                input.value = '';

                                const label = input.closest('.file-input-wrapper').querySelector('.file-label');

                                if (label) label.innerHTML = '📷 Clique para selecionar uma foto<br><span style="font-size:12px;color:#94a3b8;">Formatos: JPG, PNG, WEBP</span>';

                            } else if (input.tagName === 'SELECT') {

                                input.selectedIndex = 0;

                            } else {

                                input.value = '';

                            }

                        });

                    }

                });

 

                updateIndexes();

                updateRemoveButtons();

                updateSummary();

                messageBox.className = 'alert';

                messageBox.style.display = 'none';

                messageBox.textContent = '';

            }

            function sendEmail() {

                // Coleta dados do aluno

                const studentName = document.getElementById('studentName').value.trim();

                const studentClass = document.getElementById('studentClass').value.trim();

                const collectionDate = document.getElementById('collectionDate').value;

                const collectionLocation = document.getElementById('collectionLocation').value.trim();

                const tariff = parseFloat(tariffSelect.value) || 0.92;

                const tariffLabel = tariffSelect.options[tariffSelect.selectedIndex]?.text || 'Residencial';

 

                if (!studentName || !studentClass || !collectionDate) {

                    showMessage('⚠️ Preencha Nome, Turma e Data antes de enviar.', 'error');

                    return;

                }

 

                // Coleta equipamentos

                const sections = document.querySelectorAll('.equipment-section');

                const equipamentos = [];

                let hasError = false;

 

                sections.forEach((section, idx) => {

                    const num = idx + 1;

                    const name = document.getElementById(`equip_name_${num}`);

                    const power = document.getElementById(`equip_power_${num}`);

                    const qty = document.getElementById(`equip_quantity_${num}`);

                    const time = document.getElementById(`equip_time_${num}`);

                    const type = document.getElementById(`equip_type_${num}`);

                    const brand = document.getElementById(`equip_brand_${num}`);

                    const obs = document.getElementById(`equip_obs_${num}`);

 

                    const nameVal = name ? name.value.trim() : '';

                    const powerVal = parseFloat(power?.value) || 0;

                    const qtyVal = parseFloat(qty?.value) || 1;

                    const timeVal = parseFloat(time?.value) || 0;

                    const typeVal = type ? type.value : '';

 

                    if (!nameVal) {

                        showMessage(`⚠️ Equipamento ${num}: preencha o Nome.`, 'error');

                        hasError = true;

                        return;

                    }

                    if (powerVal <= 0) {

                        showMessage(`⚠️ Equipamento ${num}: Potência deve ser maior que zero.`, 'error');

                        hasError = true;

                        return;

                    }

                    if (timeVal < 0) {

                        showMessage(`⚠️ Equipamento ${num}: Tempo não pode ser negativo.`, 'error');

                        hasError = true;

                        return;

                    }

                    if (!typeVal) {

                        showMessage(`⚠️ Equipamento ${num}: selecione o Tipo.`, 'error');

                        hasError = true;

                        return;

                    }

 

                    const consumo = (powerVal * timeVal * qtyVal) / 1000;

                    const custo = consumo * tariff;

 

                    equipamentos.push({

                        nome: nameVal,

                        marca: brand ? brand.value.trim() : '',

                        potencia: powerVal,

                        quantidade: qtyVal,

                        tempo: timeVal,

                        tipo: typeVal,

                        consumo: consumo,

                        custo: custo,

                        observacoes: obs ? obs.value.trim() : ''

                    });

                });

 

                if (hasError) return;

                if (equipamentos.length === 0) {

                    showMessage('⚠️ Adicione pelo menos um equipamento.', 'error');

                    return;

                }

 

                // ===== CONSTRÓI O CORPO DO E-MAIL =====

                let totalConsumo = 0;

                let totalCusto = 0;

                let tableRows = '';

 

                equipamentos.forEach((eq, i) => {

                    totalConsumo += eq.consumo;

                    totalCusto += eq.custo;

                    tableRows += `

                        ${i+1}. ${eq.nome} | ${eq.potencia}W | ${eq.quantidade}x | ${eq.tempo}h/dia | ${eq.consumo.toFixed(3)} kWh | R$ ${eq.custo.toFixed(2)}

                    `;

                });

 

                const subject = encodeURIComponent(`🔍 Caça aos Equipamentos - ${studentName} (${studentClass})`);

                const body = encodeURIComponent(`

        ===================================================

        🔍 RELATÓRIO - CAÇA AOS EQUIPAMENTOS ELÉTRICOS

        ===================================================

 

        📝 DADOS DO ALUNO

        ─────────────────────────────────────────────

        Nome: ${studentName}

        Turma: ${studentClass}

        Data da coleta: ${collectionDate}

        Local: ${collectionLocation || 'Não informado'}

        Tarifa aplicada: ${tariffLabel} — R$ ${tariff.toFixed(2)} / kWh

 

        📋 EQUIPAMENTOS CADASTRADOS (${equipamentos.length})

        ─────────────────────────────────────────────

        # | Aparelho | Pot(W) | Qtd | Tempo(h/dia) | Consumo(kWh) | Custo(R$)

        ─────────────────────────────────────────────

        ${tableRows}

        ─────────────────────────────────────────────

        TOTAL GERAL: ${totalConsumo.toFixed(3)} kWh | R$ ${totalCusto.toFixed(2)}

        ===================================================

 

        📌 Observações: Os dados foram coletados durante a atividade prática.

 

        🔗 Enviado pelo sistema da Escola.

        `);

 

                window.location.href = `mailto:thyago.moncao@educacao.mg.gov.br?subject=${subject}&body=${body}`;

 

                showMessage(`✅ E-mail aberto com ${equipamentos.length} equipamentos! Basta clicar em "Enviar".`, 'success');

            }
